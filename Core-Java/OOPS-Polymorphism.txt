// Polymorphism

// Dynamic Method Dispatch - which method will override is decided at run time and is called dynamic method dispatch

class A {
    public void show() {
        System.out.println("in A show");
    }
}

class B extends A {
    public void show() {
        System.out.println("in B show");
    }
}

class C extends B {
    public void show() {
        System.out.println("in C show");
    }
}

public class Hello {

    public static void main(String args[]) throws ClassNotFoundException {
        
        // example of dynamic method dispatch
        A obj = new A();
        obj.show();

        obj = new B();
        obj.show();

        obj = new C();
        obj.show();


    }
}
